#include <stdio.h>

int main() {
    int i,n;
    int ele;
    int pos;
    printf("enter the number of elemnts you want to enter\n");
    scanf("%d",&n);
    int x[n];
    for(i=1;i<=n;i++){
        printf("enter elemnt %d \n",i);
        scanf("%d",&x[i]);
    }
    printf("enter the element that you want to search for\n");
    scanf("%d",&ele);
    for(i=1;i<=n;i++){
        if(x[i]==ele){
            pos=i;
            break;
        }
    }
    printf("the elemnt that you enter is in position %d\n",pos);
    return 0;
}
