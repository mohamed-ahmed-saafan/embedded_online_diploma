#include <stdio.h>

int main() {
   char str[100];
   char ele;
   int count;
   printf("enter a string\n");
   fgets(str,100,stdin);
   printf("enter the character you want to search for\n");
   scanf("%c",&ele);
   for(int i=0;i<100;i++){
       if(str[i]==ele){
           count++;
           
       }
   }
   printf("the char %c is repeated %d times \n",ele,count);
    return 0;
}
